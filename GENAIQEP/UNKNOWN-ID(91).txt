using System;
using NUnit.Framework;
using OpenQA.Selenium;
using OpenQA.Selenium.Chrome;
using OpenQA.Selenium.Support.UI;

namespace MyAutomationTests
{
    // FIX 1: Ensure this class exists only once in the project/snippet.
    // (Remove the duplicated second definition; if it must exist, rename it and isolate it in a different file/namespace.)

    public class LoginTest
    {
        private IWebDriver? driver;
        private WebDriverWait? wait;

        [SetUp]
        public void Setup()
        {
            // FIX 2 (Configuration): Make base URL configurable so tests target the correct environment
            // where spreadsheet data model generation may be enabled.
            var baseUrl = Environment.GetEnvironmentVariable("APP_BASE_URL")
                          ?? "https://example.com"; // fallback for local runs

            var options = new ChromeOptions();

            // FIX 3 (Configuration/CI stability): Headless support for CI runners.
            if ((Environment.GetEnvironmentVariable("HEADLESS") ?? "true").Equals("true", StringComparison.OrdinalIgnoreCase))
            {
                options.AddArgument("--headless=new");
            }

            options.AddArgument("--no-sandbox");
            options.AddArgument("--disable-dev-shm-usage");

            driver = new ChromeDriver(options);
            driver.Manage().Timeouts().PageLoad = TimeSpan.FromSeconds(60);

            // FIX 4 (Code issue): Add explicit wait for reliable synchronization
            wait = new WebDriverWait(driver, TimeSpan.FromSeconds(15));

            driver.Navigate().GoToUrl($"{baseUrl.TrimEnd('/')}/login");
        }

        [Test]
        public void TestLogin()
        {
            if (driver == null || wait == null)
                Assert.Fail("WebDriver not initialized.");

            // FIX 5 (Code issue): Wait for elements to exist/be interactable to avoid flakiness
            wait.Until(d => d.FindElement(By.Id("username")).Displayed);

            driver.FindElement(By.Id("username")).Clear();
            driver.FindElement(By.Id("username")).SendKeys(Environment.GetEnvironmentVariable("TEST_USERNAME") ?? "testuser");

            driver.FindElement(By.Id("password")).Clear();
            driver.FindElement(By.Id("password")).SendKeys(Environment.GetEnvironmentVariable("TEST_PASSWORD") ?? "password");

            driver.FindElement(By.Id("loginButton")).Click();

            // FIX 6 (Code issue): Wait for welcome message; avoid immediate FindElement assertion
            var welcome = wait.Until(d => d.FindElement(By.Id("welcomeMessage")));
            Assert.That(welcome.Displayed, Is.True, "Welcome message not displayed after login.");
        }

        [TearDown]
        public void Teardown()
        {
            // FIX 7 (Code issue): Null-safe teardown to avoid masking earlier failures
            try
            {
                driver?.Quit();
            }
            finally
            {
                driver?.Dispose();
                driver = null;
                wait = null;
            }
        }
    }
}

- Added `WebDriverWait` explicit waits so the test reliably reaches a post-login state and doesn’t fail due to timing.  
- Improved teardown with null checks + dispose so failures in setup don’t cascade into misleading teardown errors.
