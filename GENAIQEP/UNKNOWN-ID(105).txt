using System;
using System.IO;
using NUnit.Framework;
using NUnit.Framework.Interfaces;
using OpenQA.Selenium;
using OpenQA.Selenium.Chrome;
using OpenQA.Selenium.Support.UI;

namespace MyAutomationTests
{
    [TestFixture]
    public class LoginTest
    {
        private IWebDriver? driver;
        private WebDriverWait? wait;

        // CHANGE: environment-driven configuration to allow ETS-Prod execution
        private string BaseUrl => Environment.GetEnvironmentVariable("ETS_BASE_URL")
                                  ?? "https://example.com"; // fallback for local runs

        private string Username => Environment.GetEnvironmentVariable("ETS_USERNAME")
                                   ?? "user@example.com";

        private string Password => Environment.GetEnvironmentVariable("ETS_PASSWORD")
                                   ?? "Password123";

        [SetUp]
        public void Setup()
        {
            var options = new ChromeOptions();

            // CHANGE: optional headless for CI stability (toggle via env)
            if ((Environment.GetEnvironmentVariable("ETS_HEADLESS") ?? "false")
                .Equals("true", StringComparison.OrdinalIgnoreCase))
            {
                options.AddArgument("--headless=new");
            }

            driver = new ChromeDriver(options);

            // CHANGE: explicit waits to reduce flakiness and reveal real functional failures
            wait = new WebDriverWait(driver, TimeSpan.FromSeconds(30));

            driver.Navigate().GoToUrl($"{BaseUrl.TrimEnd('/')}/login");
        }

        [Test]
        public void TestLogin()
        {
            if (driver == null || wait == null)
                throw new InvalidOperationException("WebDriver not initialized.");

            // CHANGE: wait for elements instead of immediate FindElement calls
            wait.Until(d => d.FindElement(By.Id("username"))).SendKeys(Username);
            driver.FindElement(By.Id("password")).SendKeys(Password);
            driver.FindElement(By.Id("loginButton")).Click();

            // CHANGE: wait for post-login indicator (prevents timing-based false failures)
            var welcome = wait.Until(d => d.FindElement(By.Id("welcomeMessage")));
            Assert.That(welcome.Displayed, Is.True);
        }

        [TearDown]
        public void Teardown()
        {
            if (driver == null)
                return;

            // CHANGE: capture diagnostics on failure to address RCA "Insufficient Data"
            try
            {
                if (TestContext.CurrentContext.Result.Outcome.Status == TestStatus.Failed)
                {
                    Directory.CreateDirectory("artifacts");

                    // Screenshot
                    if (driver is ITakesScreenshot ss)
                    {
                        var shot = ss.GetScreenshot();
                        var screenshotPath = Path.Combine("artifacts", $"{TestContext.CurrentContext.Test.Name}_screenshot.png");
                        shot.SaveAsFile(screenshotPath);
                        TestContext.AddTestAttachment(screenshotPath);
                    }

                    // Page source
                    var htmlPath = Path.Combine("artifacts", $"{TestContext.CurrentContext.Test.Name}_page.html");
                    File.WriteAllText(htmlPath, driver.PageSource);
                    TestContext.AddTestAttachment(htmlPath);

                    // Browser console logs (works best when driver supports logs)
                    try
                    {
                        var logs = driver.Manage().Logs.GetLog(LogType.Browser);
                        var logPath = Path.Combine("artifacts", $"{TestContext.CurrentContext.Test.Name}_browser.log");
                        File.WriteAllLines(logPath, logs.Select(l => $"{l.Timestamp} {l.Level} {l.Message}"));
                        TestContext.AddTestAttachment(logPath);
                    }
                    catch
                    {
                        // Ignore if log collection isn't supported in the current driver setup
                    }
                }
            }
            finally
            {
                // CHANGE: robust cleanup even when failures occur
                driver.Quit();
                driver.Dispose();
            }
        }
    }
}


Changes Made:  
- Removed reliance on hard-coded URL and credentials by adding environment-variable configuration (`ETS_BASE_URL`, `ETS_USERNAME`, `ETS_PASSWORD`) so tests can run against ETS-Prod and the correct user/tenant.  
- Added WebDriverWait and replaced immediate `FindElement` calls with waited element retrieval to reduce flakiness and prevent timing issues from masking real functional errors.  
- Added failure diagnostics in `TearDown`: screenshot, page source, and (best-effort) browser console logs to capture evidence when the real “Agents generation” flow fails (supports moving RCA from “Unknown/Insufficient Data” to actionable root cause).  
- Hardened driver lifecycle with null checks and `try/finally` to ensure teardown always runs cleanly.  
- Important structural fix (must be applied in the repo): delete the duplicated `LoginTest` class definition from the codebase to avoid compilation/test discovery ambiguity and missing/duplicate executions.
