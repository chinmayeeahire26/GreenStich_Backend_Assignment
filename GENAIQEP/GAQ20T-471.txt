    driver.get("http://example.com/login");
    driver.findElement(By.id("username")).sendKeys("user");
    driver.findElement(By.id("password")).sendKeys("pass");
    driver.findElement(By.id("loginButton")).click();
    
    // Added assertion to verify successful login
    String expectedUrl = "http://example.com/dashboard"; // Expected result
    assert driver.getCurrentUrl().equals(expectedUrl) : "Login failed"; // Validation fix
    
    driver.quit();
}

// File: FlipkartLoginTest.java
public void loginToFlipkart() {
    System.setProperty("webdriver.chrome.driver", "/path/to/chromedriver"); // Configuration fix
    WebDriver driver = new ChromeDriver();
    driver.get("http://flipkart.com");
    driver.findElement(By.xpath("//button[contains(text(),'Login')]")).click();
    driver.findElement(By.id("email")).sendKeys("user@example.com");
    driver.findElement(By.id("password")).sendKeys("password");
    driver.findElement(By.xpath("//button[contains(text(),'Submit')]")).click();
    
    // Added verification logic for login success
    WebElement profileElement = driver.findElement(By.id("profileName"));
    assert profileElement.isDisplayed() : "Login failed"; // Validation fix
    
    // Added error handling for pop-up closure
    try {
        WebElement popUpCloseButton = driver.findElement(By.xpath("//button[contains(text(),'Close')]"));
        if (popUpCloseButton.isDisplayed()) {
            popUpCloseButton.click(); // Process fix
        }
    } catch (NoSuchElementException e) {
        System.out.println("No pop-up to close."); // Error handling
    }
    
    driver.quit();
}


Changes Made: 
- Added `System.setProperty` for WebDriver path configuration.
- Introduced verification logic using `WebElement` to ensure login success.
- Implemented error handling for potential pop-up windows using `try-catch`.
