public void createSolution(SolutionDto solutionDto) {
    boolean exists = solutionRepository.existsByNameAndOrgIdAndPortfolioId(
        solutionDto.getName(), solutionDto.getOrgId(), solutionDto.getPortfolioId());
    if (exists) {
        throw new DuplicateSolutionNameException(
            String.format("Solution name '%s' already exists in the selected portfolio and organization.",
                          solutionDto.getName()));
    }
    try {
        solutionRepository.save(solutionDto.toEntity());
    } catch (DataIntegrityViolationException e) {
        throw new DuplicateSolutionNameException(
            "Unable to create solution due to duplicate name within the portfolio and organization.");
    }
}
