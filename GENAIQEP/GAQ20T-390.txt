Integration in Controller Layer (Example):  
java
@RestController
@RequestMapping("/api/solutions")
public class SolutionController {

    @Autowired
    private FeatureStoryService featureStoryService;

    @GetMapping("/{organization}/{portfolio}/{solution}/features")
    public ResponseEntity<List<Feature>> getFeatures(
            @PathVariable String organization,
            @PathVariable String portfolio,
            @PathVariable String solution) {
        List<Feature> features = featureStoryService.getFeaturesBySolution(organization, portfolio, solution);
        if (features.isEmpty()) {
            return ResponseEntity.noContent().build();
        }
        return ResponseEntity.ok(features);
    }

    @GetMapping("/features/{featureId}/stories")
    public ResponseEntity<List<Story>> getStories(@PathVariable Long featureId) {
        List<Story> stories = featureStoryService.getStoriesByFeatureId(featureId);
        if (stories.isEmpty()) {
            return ResponseEntity.noContent().build();
        }
        return ResponseEntity.ok(stories);
    }
}
