public void loginToFlipkart() {
    driver.get("https://www.flipkart.com");
    driver.findElement(By.cssSelector("input._2IX_2- VJZDxU")).sendKeys("username");
    driver.findElement(By.cssSelector("input._2IX_2- _3mctLh VJZDxU")).sendKeys("password");
    driver.findElement(By.cssSelector("button._2KpZ6l _2HKlqd _3AWRsL")).click();

    // Added UI verification to detect unwanted borders or lines after login
    WebElement pageBody = driver.findElement(By.cssSelector("body"));
    String borderBottom = pageBody.getCssValue("border-bottom");
    Assert.assertTrue(borderBottom.equals("none") || borderBottom.equals("0px"), 
        "Unexpected bottom border found on the page after login.");

    List<WebElement> bottomLines = driver.findElements(By.cssSelector(".bottom-line"));
    Assert.assertTrue(bottomLines.isEmpty() || bottomLines.stream().allMatch(e -> !e.isDisplayed()),
        "Unwanted bottom line element is visible on the page.");
}
