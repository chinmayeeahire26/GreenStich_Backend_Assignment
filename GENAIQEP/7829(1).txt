public class FlipkartLoginTest {
    public void loginToFlipkart() {
        WebDriver driver = null;
        try {
            // Initialize WebDriver with error handling
            driver = new ChromeDriver();
            driver.get("https://www.flipkart.com");

            // Use environment variables or secure config for credentials
            String username = System.getenv("FLIPKART_USERNAME");
            String password = System.getenv("FLIPKART_PASSWORD");
            if (username == null || password == null) {
                throw new IllegalStateException("Missing login credentials in environment variables");
            }

            driver.findElement(By.cssSelector("input[type='text']")).sendKeys(username);
            driver.findElement(By.cssSelector("input[type='password']")).sendKeys(password);
            driver.findElement(By.cssSelector("button[type='submit']")).click();

            // Verification logic to confirm successful login
            boolean loginSuccess = driver.findElements(By.cssSelector("selector-for-logged-in-element")).size() > 0;
            if (!loginSuccess) {
                throw new RuntimeException("Login failed: verification element not found");
            }
        } catch (Exception e) {
            // Log and handle exceptions appropriately
            e.printStackTrace();
            throw e;
        } finally {
            if (driver != null) {
                driver.quit();
            }
        }
    }
}
