import java.net.URL;
import java.net.URLDecoder;
import java.nio.charset.StandardCharsets;
import java.util.regex.Pattern;

public void createSolution(SolutionRequest request) throws InvalidInputException {
    String rawJiraApiUrl = request.getJiraBaseUrl();

    // Decode URL to handle encoded characters
    String decodedUrl = URLDecoder.decode(rawJiraApiUrl, StandardCharsets.UTF_8.name());

    // Validate URL format
    if (!isValidUrl(decodedUrl)) {
        throw new InvalidInputException("Invalid Jira Base URL provided.");
    }

    URL url = new URL(decodedUrl); // Properly formed URL

    // Proceed with API call
}

private boolean isValidUrl(String url) {
    String urlRegex = "^(https?)://[w.-]+(:d+)?(/.*)?$";
    return Pattern.matches(urlRegex, url);
}
