import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;
import org.testng.Assert;
import org.testng.annotations.Test;

import java.time.Duration;

public class LllllLoginTest {

    @Test
    public void loginAndValidatePayLiabilityReportOverpaymentPTD() {
        // CHANGE: Externalize driver path for portability (CI/local).
        System.setProperty("webdriver.chrome.driver",
                System.getProperty("CHROMEDRIVER_PATH", "path/to/chromedriver"));

        // CHANGE: Add options hook for stability (headless/downloads can be added later as needed).
        ChromeOptions options = new ChromeOptions();
        WebDriver driver = new ChromeDriver(options);

        // CHANGE: Add explicit waits to avoid Angular/Material timing issues.
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(20));

        // CHANGE: Externalize URL/credentials to avoid hard-coding and reduce env/user dependency.
        String baseUrl = System.getProperty(
                "PORTAL_URL",
                "https://itg-lllld-d-lllld-portal.azurewebsites.net/login?redirect=%2F"
        );
        String username = System.getProperty("PORTAL_USER", "lllld_test_r03@lllldlllldlllld.com");
        String password = System.getProperty("PORTAL_PASS", "Microsoft!1");

        try {
            driver.get(baseUrl);

            WebElement usernameField = wait.until(
                    ExpectedConditions.visibilityOfElementLocated(By.name("email")));
            WebElement passwordField = wait.until(
                    ExpectedConditions.visibilityOfElementLocated(By.name("password")));

            usernameField.sendKeys(username);
            passwordField.sendKeys(password);

            WebElement loginButton = wait.until(ExpectedConditions.elementToBeClickable(
                    By.xpath("/html/body/app-root/app-login/div/mat-card/mat-card-content/form/button")
            ));
            loginButton.click();

            // CHANGE: Basic checkpoint that login/navigation shell is present.
            wait.until(ExpectedConditions.presenceOfElementLocated(
                    By.xpath("/html/body/app-root/app-layout")));

            // CHANGE: Prevent false-green runs. Real validation for the PTD column is not present
            // in the provided code and cannot be implemented without correct report page locators.
            Assert.fail(
                    "Missing implementation: navigate to Pay > Liability Report_ITG and validate " +
                    "'Over Payment Liability PTD' values. Add stable selectors (prefer data-testid) " +
                    "and assert displayed PTD against expected source (API/DB/export)."
            );

        } catch (Exception e) {
            // CHANGE: Make failures visible/actionable in CI rather than printStackTrace only.
            Assert.fail("UI automation failed before PTD validation: " + e.getMessage(), e);
        } finally {
            driver.quit();
        }
    }
}

- Added a deliberate Assert.fail to prevent a false pass since the test still does not (and cannot, with provided info) navigate to the report and validate the “Over Payment Liability PTD” column.
