import org.apache.commons.lang3.StringUtils;
import org.owasp.encoder.Encode;

public User loginUser() {
    Authentication auth = SecurityContextHolder.getContext().getAuthentication();
    String username = auth.getName();

    // Input validation: check for null, empty, and length constraints
    if (StringUtils.isBlank(username) || username.length() > 50) {
        throw new IllegalArgumentException("Invalid username");
    }

    // Input sanitization: encode to prevent injection if used in output (defensive)
    String safeUsername = Encode.forHtml(username);

    // Use sanitized username for querying - assuming repository method expects raw username,
    // so validate and sanitize input but query with validated input
    User user = userRepository.findByUsername(username.trim());

    return user;
}


Changes Made:  
- Added validation for blank and length constraints on username.  
- Added defensive encoding using OWASP Encoder.  
- Trimmed username before querying.  
- Threw exception on invalid input.

public void loginToFlipkart(String username, String password) {
    WebElement usernameField = driver.findElement(By.id("username"));
    WebElement passwordField = driver.findElement(By.id("password"));

    // Input validation: simple regex to allow alphanumeric usernames only
    if (username == null || !username.matches("^[a-zA-Z0-9._-]{3,50}$")) {
        throw new IllegalArgumentException("Invalid username format");
    }
    if (password == null || password.length() < 6 || password.length() > 50) {
        throw new IllegalArgumentException("Invalid password length");
    }

    usernameField.clear();
    usernameField.sendKeys(username); // validated input
    passwordField.clear();
    passwordField.sendKeys(password); // validated input

    driver.findElement(By.id("loginButton")).click();

    // Verification: wait and check for login success indicator
    try {
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));
        wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("userProfileIcon")));
        System.out.println("Login successful");
    } catch (TimeoutException e) {
        System.err.println("Login failed or user profile icon not found");
        // Additional error handling or test failure logic here
    } catch (Exception e) {
        System.err.println("Unexpected error during login: " + e.getMessage());
        // Handle other exceptions gracefully
    }
}
