public void loginToSailpoint() {
    WebDriver driver = null;
    try {
        driver = new ChromeDriver();
        driver.get(System.getenv("SAILPOINT_LOGIN_URL")); // URL is now configurable
        WebElement username = driver.findElement(By.id("username"));
        if (username != null) { // Validate element presence
            username.sendKeys(System.getenv("SAILPOINT_USERNAME")); // Credentials are now configurable
        }
        WebElement password = driver.findElement(By.id("password"));
        if (password != null) { // Validate element presence
            password.sendKeys(System.getenv("SAILPOINT_PASSWORD")); // Credentials are now configurable
        }
        WebElement loginButton = driver.findElement(By.id("loginButton"));
        if (loginButton != null) { // Validate element presence
            loginButton.click();
        }
    } catch (NoSuchElementException e) {
        System.out.println("Element not found: " + e.getMessage()); // Exception handling
    } catch (WebDriverException e) {
        System.out.println("WebDriver error: " + e.getMessage()); // Exception handling
    } finally {
        if (driver != null) {
            driver.quit(); // Ensure driver quits to free resources
        }
    }
}


Changes Made: 
- Added try-catch blocks to handle potential exceptions during WebDriver operations.
- Validated the presence of elements before interacting with them to avoid `NoSuchElementException`.
- Made credentials and URLs configurable using environment variables to avoid hardcoding.
- Implemented logging for exceptions to aid in debugging and analysis.
