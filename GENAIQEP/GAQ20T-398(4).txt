public void updateLLM(String llmId, LLMUpdatePayload payload) {
    // Defensive: Validate inputs
    if (llmId == null || payload == null) {
        throw new IllegalArgumentException("LLM id and payload must not be null");
    }

    // Retrieve the specific LLM to update
    Optional<LLM> llmOptional = llmRepository.findById(llmId);
    if (!llmOptional.isPresent()) {
        throw new ResourceNotFoundException("LLM with id " + llmId + " not found");
    }

    // Build a fresh payload scoped strictly to the selected LLM
    Map<String, Object> scopedPayload = new HashMap<>(payload.toMap());
    scopedPayload.put("llmId", llmId); // Explicitly bind payload to selected LLM

    // Apply update with confirmed LLM id; do not reuse shared/global maps
    llmRepository.updateLLMById(llmId, scopedPayload);
}
